let currentStep = 1; const totalSteps = 3; let configData = {}; let selectedTribunals = new Set(); document.addEventListener('DOMContentLoaded', function() { console.log('🚀 Monitor Jurídico Config carregando...'); // Carregar dados existentes se houver loadExistingData(); // Configurar eventos dos tribunais setupTribunalSelection(); // Aplicar máscaras applyMasks(); // Configurar animações setupAnimations(); console.log('✅ Monitor Config inicializado'); }); function loadExistingData() { const savedConfig = localStorage.getItem('monitorConfig'); if (savedConfig) { configData = JSON.parse(savedConfig); populateForm(configData); } } function populateForm(data) { Object.keys(data).forEach(key => { const element = document.getElementById(key); if (element) { element.value = data[key]; } }); if (data.tribunals) { selectedTribunals = new Set(data.tribunals); updateTribunalSelection(); } } function setupTribunalSelection() { document.querySelectorAll('.tribunal-card').forEach(card => { card.addEventListener('click', function() { const tribunal = this.dataset.tribunal; if (selectedTribunals.has(tribunal)) { selectedTribunals.delete(tribunal); this.classList.remove('selected'); } else { selectedTribunals.add(tribunal); this.classList.add('selected'); } }); }); } function updateTribunalSelection() { document.querySelectorAll('.tribunal-card').forEach(card => { const tribunal = card.dataset.tribunal; if (selectedTribunals.has(tribunal)) { card.classList.add('selected'); } else { card.classList.remove('selected'); } }); } function applyMasks() { // Máscara CPF document.getElementById('cpf').addEventListener('input', function(e) { let value = e.target.value.replace(/\D/g, ''); value = value.replace(/(\d{3})(\d)/, '$1.$2'); value = value.replace(/(\d{3})(\d)/, '$1.$2'); value = value.replace(/(\d{3})(\d{1,2})$/, '$1-$2'); e.target.value = value; }); // Máscara Telefone document.getElementById('phone').addEventListener('input', function(e) { let value = e.target.value.replace(/\D/g, ''); value = value.replace(/(\d{2})(\d)/, '($1) $2'); value = value.replace(/(\d{5})(\d)/, '$1-$2'); e.target.value = value; }); // Máscara OAB Number document.getElementById('oabNumber').addEventListener('input', function(e) { let value = e.target.value.replace(/\D/g, ''); e.target.value = value; }); } function setupAnimations() { // Adicionar animação de entrada para os elementos const observer = new IntersectionObserver((entries) => { entries.forEach(entry => { if (entry.isIntersecting) { entry.target.style.animationDelay = Math.random() * 0.3 + 's'; entry.target.classList.add('animate-in'); } }); }); document.querySelectorAll('.tribunal-card').forEach(card => { observer.observe(card); }); } function nextStep() { if (!validateCurrentStep()) { return; } if (currentStep < totalSteps) { saveCurrentStepData(); currentStep++; updateStepDisplay(); if (currentStep === 3) { showDataReview(); } } else { finishConfiguration(); } } function previousStep() { if (currentStep > 1) { currentStep--; updateStepDisplay(); } } function validateCurrentStep() { switch (currentStep) { case 1: return validateStep1(); case 2: return validateStep2(); case 3: return validateStep3(); default: return true; } } function validateStep1() { const requiredFields = ['oabNumber', 'oabState', 'lawyerType', 'fullName', 'cpf', 'phone', 'email']; let isValid = true; requiredFields.forEach(fieldId => { const field = document.getElementById(fieldId); field.classList.remove('error'); if (!field.value.trim()) { field.classList.add('error'); isValid = false; } }); if (!isValid) { showNotification('Por favor, preencha todos os campos obrigatórios.', 'error'); } return isValid; } function validateStep2() { if (selectedTribunals.size === 0) { showNotification('Selecione pelo menos um tribunal para monitoramento.', 'warning'); return false; } return true; } function validateStep3() { const acceptTerms = document.getElementById('acceptTerms').checked; const acceptPrivacy = document.getElementById('acceptPrivacy').checked; if (!acceptTerms || !acceptPrivacy) { showNotification('Você deve aceitar os termos de uso e a política de privacidade para continuar.', 'warning'); return false; } return true; } function saveCurrentStepData() { switch (currentStep) { case 1: configData.oabNumber = document.getElementById('oabNumber').value; configData.oabState = document.getElementById('oabState').value; configData.lawyerType = document.getElementById('lawyerType').value; configData.fullName = document.getElementById('fullName').value; configData.cpf = document.getElementById('cpf').value; configData.phone = document.getElementById('phone').value; configData.email = document.getElementById('email').value; break; case 2: configData.tribunals = Array.from(selectedTribunals); break; } localStorage.setItem('monitorConfig', JSON.stringify(configData)); } function updateStepDisplay() { // Atualizar indicadores de etapa for (let i = 1; i <= totalSteps; i++) { const stepElement = document.getElementById(`step${i}`); stepElement.classList.remove('active', 'completed'); if (i < currentStep) { stepElement.classList.add('completed'); } else if (i === currentStep) { stepElement.classList.add('active'); } } // Mostrar/ocultar etapas document.querySelectorAll('.config-step').forEach((step, index) => { step.classList.add('hidden'); step.classList.remove('active'); if (index + 1 === currentStep) { step.classList.remove('hidden'); setTimeout(() => step.classList.add('active'), 100); } }); // Atualizar botões const backBtn = document.getElementById('backBtn'); const nextBtn = document.getElementById('nextBtn'); if (currentStep === 1) { backBtn.classList.add('hidden'); } else { backBtn.classList.remove('hidden'); } if (currentStep === totalSteps) { nextBtn.innerHTML = ' Finalizar Configuração'; } else { nextBtn.innerHTML = 'Próximo '; } } function showDataReview() { const reviewContainer = document.getElementById('dataReview'); const tribunalsText = Array.from(selectedTribunals).join(', '); reviewContainer.innerHTML = `

Dados Profissionais

OAB: ${configData.oabNumber}/${configData.oabState}
Nome: ${configData.fullName}
Tipo: ${configData.lawyerType}
CPF: ${configData.cpf}
Telefone: ${configData.phone}
E-mail: ${configData.email}

Tribunais Selecionados

Tribunais: ${tribunalsText}
`; } function finishConfiguration() { // Mostrar loading const nextBtn = document.getElementById('nextBtn'); nextBtn.innerHTML = ' Processando...'; nextBtn.disabled = true; // Salvar configuração final configData.acceptTerms = true; configData.acceptPrivacy = true; configData.configuredAt = new Date().toISOString(); configData.status = 'pending'; localStorage.setItem('monitorConfig', JSON.stringify(configData)); localStorage.setItem('monitorConfigured', 'true'); // Simular processamento setTimeout(() => { showSuccessStep(); showNotification('Configuração concluída com sucesso!', 'success'); // Redirecionar para login após 5 segundos setTimeout(() => { if (confirm('Configuração concluída! Você será redirecionado para o login para acessar seu painel. Continuar?')) { window.location.href = 'login.html'; } }, 5000); }, 2000); } function showSuccessStep() { document.querySelectorAll('.config-step').forEach(step => { step.classList.add('hidden'); step.classList.remove('active'); }); const successStep = document.getElementById('configSuccess'); successStep.classList.remove('hidden'); setTimeout(() => successStep.classList.add('active'), 100); // Atualizar botões document.getElementById('backBtn').classList.add('hidden'); const nextBtn = document.getElementById('nextBtn'); nextBtn.innerHTML = ' Acessar Monitor'; nextBtn.disabled = false; nextBtn.onclick = function() { window.open('/monitor-juridico/public/index.html', '_blank'); }; // Atualizar todos os steps como completed document.querySelectorAll('.step').forEach(step => { step.classList.remove('active'); step.classList.add('completed'); }); } function showNotification(message, type = 'info') { // Criar notification toast const notification = document.createElement('div'); notification.style.cssText = ` position: fixed; top: 20px; right: 20px; background: ${type === 'success' ? 'var(--success)' : type === 'error' ? 'var(--danger)' : 'var(--warning)'}; color: white; padding: 1rem 1.5rem; border-radius: 8px; box-shadow: var(--shadow-lg); z-index: 10000; animation: slideIn 0.3s ease; `; notification.textContent = message; document.body.appendChild(notification); setTimeout(() => { notification.style.animation = 'slideOut 0.3s ease'; setTimeout(() => notification.remove(), 300); }, 4000); } function goToDashboard() { window.location.href = 'dashboard.html'; } function showTerms() { const modal = document.createElement('div'); modal.className = 'modal-bg'; modal.innerHTML = `

Termos de Uso

Ao aceitar estes termos, você autoriza o Ali Software Jurídico a:

  • Utilizar suas credenciais da OAB para acessar informações públicas de processos
  • Monitorar automaticamente os tribunais selecionados
  • Enviar notificações sobre movimentações processuais
  • Armazenar dados dos processos para fins de acompanhamento

Todos os dados são tratados com segurança e conforme a LGPD.

`; document.body.appendChild(modal); } function showPrivacy() { const modal = document.createElement('div'); modal.className = 'modal-bg'; modal.innerHTML = `

Política de Privacidade

O Ali Software Jurídico está comprometido com a proteção dos seus dados pessoais:

  • Coletamos apenas dados necessários para o funcionamento do serviço
  • Não compartilhamos dados com terceiros sem autorização
  • Utilizamos criptografia para proteger informações sensíveis
  • Você pode solicitar exclusão dos dados a qualquer momento
  • Cumprimos todas as diretrizes da LGPD

Para mais informações, entre em contato conosco.

`; document.body.appendChild(modal); } // CSS para animações const style = document.createElement('style'); style.textContent = ` @keyframes slideIn { from { transform: translateX(100%); opacity: 0; } to { transform: translateX(0); opacity: 1; } } @keyframes slideOut { from { transform: translateX(0); opacity: 1; } to { transform: translateX(100%); opacity: 0; } } .animate-in { animation: fadeInUp 0.6s ease forwards; } @keyframes fadeInUp { from { opacity: 0; transform: translateY(20px); } to { opacity: 1; transform: translateY(0); } } `; document.head.appendChild(style);
